def get_var_file(varno:int, lfdn=False, df=False):
    basedir = '../../data/freetext'
    long_vars = sorted(
        [x for x in os.listdir(basedir) if 'long' in x],
          key=lambda x:int(x.split('_')[1]))
    this_file = [x for x in long_vars if int(x.split('_')[1]) == varno][0]
    print(this_file)
    if df:
        return pd.read_csv(f'{basedir}/{this_file}')
    elif not lfdn:
        return pd.read_csv(f'{basedir}/{this_file}')[f'v_{varno}'].values
    else:
        return pd.read_csv(f'{basedir}/{this_file}').values

def create_overview_df(df, longvarlist, depvarlist):
    varx = ['v_' + str(x) for x in depvarlist + longvarlist]
    overview = pd.DataFrame(columns=['lfdn','problem','response'])
    for idx, var in enumerate(varx[0:int(len(varx)/2)]):
        overview = overview.append(
            df[['lfdn'] + [var, varx[idx+5]]].rename({var:'problem',varx[idx+5]:'response'}, axis=1))

def prepare_nfr_overview(df, longvarlist):
    varx = ['v_' + str(x) for x in longvarlist]
    overview = pd.DataFrame(columns=['lfdn','documentation'])
    for idx, var in enumerate(varx):
        overview = overview.append(
            df[['lfdn', var]].rename({var:'documentation'}, axis=1))
    overview['documentation_lower'] = [x.lower() for x in overview['documentation']]
    return overview[~overview['documentation'].isin({'NotShown','NotAnswered','NotApplicable'})
            ].sort_values(['documentation_lower','lfdn']).drop('documentation_lower', axis=1)
